import numpy as np
import matplotlib.pyplot as plt
from scipy.optimize import fsolve
import tkinter as tk
from tkinter import messagebox

# This program is designed to take in fundimental geometry reated to a cycloidal gearbox (radii, eccentriciy, and pin counts) and output the optimal profile
# The contact points when the eccentricity is parallel to the x axis are taken and geometrically interpreted to find where the forces converge. 
# This allows for an accurate force balance that can then be used to solve for the herzian contact forces to hollistically spec out materials.


# I fear that if any values were to change, it would drasitcally change the accuracy of this program.
# Edit at your own risk

class Cycloid:
    def __init__(self, R_1 , Rr_1 , e , Z1 , Z2 , B , OR, OP):
        self.R_1, self.Rr_1 = R_1, Rr_1
        self.e, self.B, self.OR, self.OP = e , B , OR, OP
        self.Z1, self.Z2 = Z1, Z2
        #test guesses
        self.k0, self.l0 = 0, np.pi
        self.deg_shift = 2* np.pi / self.Z2 
        #discrete points ; better for matrices but probably pretty slow
        self.x = np.arange(0, 2*np.pi, 0.0001)

        #run once for entire scope
        self.contacts = Cycloid.Closest_Pair(self)

        #material properties
        self.E1 = 69.5
        self.E2 = 69.5
        self.v1 = 0.33
        self.v2 = 0.33

        Yield = 275

        #moment arm in N-m
        m = 20
        moment = m *1000   # N-mm

        self.forces = Cycloid.force_balance(self, moment)
        max_stress = Cycloid.Contact_Stresses(self)

        if max_stress/Yield < 0:
            passedYN = "FAILED"
        else:
            passedYN = "PASSED"

        root = tk.Tk()
        root.withdraw()  # hide the main window
        messagebox.showinfo("Results", 
            f"Max force at inner pins: {round(self.forces[2],2)}N located at pin: {round(self.forces[3],2)}"
            f"\nwith a max VM stress of: {round(max_stress,2)} MPa"
            f"\n{passedYN}"
            f"\nPin 0 starts on rightmost x intercept and spans CCW")
        
        print('\n\n\nMax force at outer pins:' , self.forces[0] , 'N which is at pin:' , self.forces[1])


    def geometry(self):
        #contact angle
        Psi = np.arctan( np.sin(self.Z1 * self.x)/ (np.cos(self.Z1 * self.x) - self.Rr_1 /(self.e*(self.Z1+1))))
        #profile calcs from literature
        Cx = self.Rr_1*np.cos(self.x) - self.R_1*np.cos(self.x + Psi) - self.e*np.cos(self.x*(self.Z1+1))
        Cy = -self.Rr_1*np.sin(self.x) + self.R_1*np.sin(self.x + Psi) + self.e*np.sin(self.x*(self.Z1+1))
        return Cx , Cy
    
    def rollers(self):
        #each roller gets 1000 data points
        size = (self.Z2 + 1 ) * 360
        shell = np.zeros((size, 2))
        centers = Cycloid.center_of_rollers(self)
        theta = np.linspace(0, 2 * np.pi, 360, endpoint=False)
        #get geometry for each roller
        for i in range(self.Z2):
            # circumference points: sweep theta around each roller center
            start = i * 360
            shell[start:start+360, 0] = self.R_1 * np.cos(theta) + centers[i, 0]
            shell[start:start+360, 1] = self.R_1 * np.sin(theta) + centers[i, 1]  

        return shell
    
    def center_of_rollers(self):
        centers = np.zeros((self.Z2, 2))
        for i in range(self.Z2):
            #offset from datum (center of circle cords)
            centers[i,0] = (self.Rr_1 * np.cos(i*self.deg_shift) - self.e)
            centers[i,1] = (self.Rr_1 * np.sin(i*self.deg_shift))
        return centers

    def Closest_Pair(self):
        centers = self.center_of_rollers()
        contacts = np.zeros((self.Z2, 3))

        for i in range(self.Z2):
            self.Ex = centers[i, 0]
            self.Ey = centers[i, 1]

            # coarse sweep for initial guess
            k_vals = np.linspace(0, 2*np.pi, 500)
            l_vals = np.linspace(0, 2*np.pi, 500)
            best, best_params = np.inf, (0, 0)
            for k in k_vals:
                fx, fy = self.f(k)
                for l in l_vals:
                    gx, gy = self.g(l)
                    d = (fx-gx)**2 + (fy-gy)**2
                    if d < best:
                        best, best_params = d, (k, l)

            # refine with fsolve
            def equations(params):
                k, l = params
                fx, fy = self.f(k)
                gx, gy = self.g(l)
                return [fx - gx, fy - gy]

            solution, info, ier, msg = fsolve(equations, best_params, full_output=True)
            k_sol, l_sol = solution

            if abs(info['fvec'][0]) > 2:
                pass
            else:
                # [x_val of contact] , [Y_val of contact] , [slope of contact]
                contacts[i,0:2] = np.double(self.f(solution[0]))
                contacts[i,2] = Cycloid.normals(self, k_sol)
        #print(contacts)
        return contacts
    
    def normals(self, k_sol, eps=1e-6):
        x1, y1 = self.f(k_sol - eps)
        x2, y2 = self.e*self.Z2 , 0
        #omit for now
        dx = (x2 - x1) / (2 * eps)
        dy = (y2 - y1) / (2 * eps)
        #slope to convergence
        DY = (y2 - y1) / (2)
        DX = (x2 - x1) / (2)

        return DY/DX

    def f(self, k):
        return ((self.Rr_1*np.cos(k) 
                -self.R_1*np.cos(k 
                +(np.arctan( np.sin(self.Z1 * k)/ (np.cos(self.Z1 * k) - self.Rr_1 /(self.e*(self.Z1+1))))))) 
                - self.e*np.cos(k*(self.Z1+1)) 
                ,                   
                (-self.Rr_1*np.sin(k) 
                +self.R_1*np.sin(k 
                +(np.arctan( np.sin(self.Z1 * k)/ (np.cos(self.Z1 * k) - self.Rr_1 /(self.e*(self.Z1+1))))))) 
                + self.e*np.sin(k*(self.Z1+1)))
    
    def g(self, l):
        scale = 2
        return (( (self.R_1-scale)*np.cos(l)+( self.Ex - self.e )), ( (self.R_1-scale)*np.sin(l) + self.Ey ))
    
    def bore(self):
        Cx  = self.B*np.cos(self.x) + self.e
        Cy  = self.B*np.sin(self.x)
        return Cx , Cy
    
    def output_rollers(self):
        #each roller gets 1000 data points
        size = (self.Z2 + 1 ) * 360 + 1
        Output_Rollers = np.zeros((size, 2))
        Output_pins = np.zeros((size, 2))
        theta = np.linspace(0, 2 * np.pi, 360, endpoint=False)
        step = 2 * np.pi / 8

        #moment arms
        M_inner = np.zeros((8,1))

        for k in range(8):
            # circumference points: sweep theta around each roller center
            start = k * 360
            Output_Rollers[start:start+360, 0] = self.OR * np.cos(theta) + np.cos(k*step) *(36.98) + self.e
            Output_Rollers[start:start+360, 1] = self.OR * np.sin(theta) + np.sin(k*step)*(36.98)
            Output_pins[start:start+360, 0] = self.OP * np.cos(theta) + np.cos(k*step) *(36.98)
            Output_pins[start:start+360, 1] = self.OP * np.sin(theta) + np.sin(k*step)*(36.98)
            M_inner[k] =  np.sin(k*step)*(36.98)
        #print(Output_Rollers)
        return Output_Rollers , Output_pins, M_inner

    def point_slope(self):
        # Upper and lower bounds of slope (x vals)
        xyUP = np.zeros((self.Z2,2))
        xyLOW = np.zeros((self.Z2,2))

        for i in range(self.Z2):
            if self.contacts[i,0] == 0:
                pass
            else:
                b = self.contacts[i,1] - self.contacts[i,0]*self.contacts[i,2]
                xyUP[i,0] = self.contacts[i,0]
                xyUP[i,1]  = self.contacts[i,2] * self.contacts[i,0] + b

                xyLOW[i,0] = self.e * self.Z2
                xyLOW[i,1] = 0

        return np.array([xyUP, xyLOW])

    def Output_Reactions(self):
        rad = 36.98
        num = 8
        centers = np.zeros((num, 2))
        reactions = np.zeros((num, 2))
        deg_shift = 2 * np.pi / num
        for i in range(8):
            #offset from datum (center of circle cords)
            # X cords
            centers[i,0] = (rad * np.cos(i*deg_shift))
            reactions[i,0] = centers[i,0] - self.OP
            # Y cords
            centers[i,1] = (rad * np.sin(i*deg_shift))
            reactions[i,1] = centers[i,1]

        X_OR = centers[:,0] , reactions[:,0]
        Y_OR = centers[:,1] , reactions[:,1]
        return X_OR, Y_OR

    def force_balance(self, T_input):
        T_output  = T_input * self.Z1
        m_outer = self.contacts[:, 2]          # normal slopes
        x_outer = self.contacts[:, 0]          # contact x
        y_outer = self.contacts[:, 1]          # contact y
        OR, OP, M_inner = Cycloid.output_rollers(self) #moment arm for inner rollers (M_inner)
        cos_phi = 1 / np.sqrt(m_outer**2 + 1)

        r_outer = np.abs(m_outer * x_outer - y_outer) / np.sqrt(m_outer**2 + 1)
        r_inner = M_inner

        # Outer Pins accounting for two disks
        F_outer = T_input * r_outer / np.sum(r_outer**2) / 2 
        F_inner = T_output * r_inner / np.sum(r_inner**2) / 8

        max_index1 = np.argmax(F_outer)
        max_index2 = np.argmax(F_inner)

        return np.max(F_outer) , max_index1 , np.max(F_inner) , max_index2
    
    def Contact_Stresses(self):
        forces = self.forces
        F_c = np.max(forces)
        D1 = self.OR/1000
        D2 = self.OP/1000

        L = 6.35/1000

        E_e = 10**9 * ((1-(self.v1)**2) /self.E1 + (1-(self.v2)**2)/self.E2)**(-1)

        D_e = 2* (D1 * D2 / (D1 - D2))

        w =2*np.sqrt((2*F_c*E_e)/(np.pi*L*E_e))
        
        P_max = np.sqrt((2*F_c*E_e)/(np.pi*L*D_e)) / (10**6)

        Shear_max = 0.3*P_max

        Tensile_max = (1-2*(self.v1))*P_max/3
        VM = np.sqrt((Tensile_max**2) + 3*(Shear_max**2))
        return VM

    def plotted(self):
        Convergence = Cycloid.point_slope(self)
        x_limits = Convergence[0][:,0] , Convergence[1][:,0]
        y_extended = Convergence[0][:,1] , Convergence[1][:,1]

        [Cx , Cy] = Cycloid.geometry(self)
        shell = Cycloid.rollers(self)
        center = Cycloid.center_of_rollers(self)
        [Bx , By] = Cycloid.bore(self)
        X_OR , Y_OR = Cycloid.Output_Reactions(self)
        OR , OP, M = Cycloid.output_rollers(self)

        plt.figure(figsize=(8, 8))

        # Set the axis range from -70 to 70
        plt.xlim(-70, 70)
        plt.ylim(-70, 70)

        plt.scatter(Bx, By, color='green', marker='.', s=0.5)
        plt.scatter(Cx, Cy, color='green', marker='.', s=0.5)
        plt.scatter(OR[:,0], OR[:,1], color='green', marker='.', s=0.5)
        plt.scatter(OP[:,0], OP[:,1], color='green', marker='.', s=0.5)

        plt.scatter(center[:,0], center[:,1], color='orange', marker='.', s=0.5) 
        plt.scatter(shell[:,0], shell[:,1], color='blue', marker='.' , s=0.5)  
        plt.scatter(self.contacts[:,0], self.contacts[:,1], color='red', marker='.', s=100) 
        plt.plot(x_limits, y_extended, color='red')
        plt.plot(X_OR, Y_OR, color='red') 

        plt.title("2D Cycloid Profile")
        plt.xlabel("X")
        plt.ylabel("Y")
        plt.legend()
        plt.grid(True)
        plt.show()
